#include <bits/stdc++.h>

using namespace std;

// Complete the makingAnagrams function below.
int makingAnagrams(string s1, string s2) {
    int a[26]={0};
    int b[26]={0};
    for(int i=0;i<s1.size();i++)
{
    a[s1[i]-'a']++;
}
for(int i=0;i<s2.size();i++){
    b[s2[i]-'a']++;
}
int ret=0;
for(int i=0;i<26;i++){
    ret=ret+abs(a[i]-b[i]);
}
return ret;

}

int main()
{
    ofstream fout(getenv("OUTPUT_PATH"));

    string s1;
    getline(cin, s1);

    string s2;
    getline(cin, s2);

    int result = makingAnagrams(s1, s2);

    fout << result << "\n";

    fout.close();

    return 0;
}
